alert组件

实例代码

var alert = new jam.cUIAlert({
    showTitle: true,
    title: '头部',
    message: '内容',
    needMask: true,
    needMaskHide: true,
    disableScroll: false,
    buttons: [
        {
            text: '你确定?',
            click: function () {
                this.hide();
            }
        },
        {
            type: 'cancel',
            text: '取消!',
            click: function () {
                this.hide();
            }
        }
    ]
});
$('#button1').on('click', function () {
    alert.show();
})
